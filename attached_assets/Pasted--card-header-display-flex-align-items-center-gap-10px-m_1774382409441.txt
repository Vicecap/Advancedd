function solve() {
    let question = document.getElementById('question').value;
    question = question.replace(/(\d)\s*x\s*(\d)/gi, '$1*$2');
    document.getElementById('question').value = question;

    const ai = document.getElementById('aiSelect').value;
    const topic = document.getElementById('topicSelect').value;
    const output = document.getElementById('output');
    const answerField = document.getElementById('answer');
    const graphWrapper = document.getElementById('graphInputWrapper');

    output.innerHTML = '';
    answerField.value = '';
    if (graphWrapper) graphWrapper.style.display = 'none';

    startLoadingDots(output);

    if (typeof chartInstance !== "undefined" && chartInstance) chartInstance.destroy();
    if (document.getElementById('graphCanvas')) Plotly.purge('graphCanvas');

    const evtSource = new EventSource(`/solve-stream?question=${encodeURIComponent(question)}&ai=${ai}&topic=${topic}`);
    let finalAnswerSet = false;

    evtSource.onmessage = e => {
        const data = JSON.parse(e.data);
        stopLoadingDots();

        if (!finalAnswerSet && (data.answer || data.solution)) {
            let ansText = data.answer || data.solution;
            ansText = ansText.replace(/Step\s*\d+:|Final Answer:|Solution:/gi, '').trim();
            answerField.value = ansText;
            finalAnswerSet = true;
        }

        if (data.paragraph) output.innerHTML += data.paragraph + '\n\n';

        if (topic === 'plot' && graphWrapper) {
            graphWrapper.style.display = 'block';
            try { plotFunction(question); } catch (err) { console.error(err); }
        }

        if (data.answer) {
            try {
                const nums = JSON.parse(data.answer);
                if (Array.isArray(nums) && graphWrapper) {
                    graphWrapper.style.display = 'block';
                    renderChart(nums);
                }
            } catch {}
        }

        if (data.done) evtSource.close();
    };
}

function decodeQuizText(text, encoding) {
    if (encoding === 'base64') { try { return atob(text); } catch { return text; } }
    if (encoding === 'url3986') { try { return decodeURIComponent(text); } catch { return text; } }                                                 return text;
}

    choicesDiv.innerHTML = '';

        [...q.incorrect_answers, q.correct_answer]

        btn.innerHTML = c;
        btn.onclick = () => handleAnswer(c, decodeQuizText(q.correct_answer, encoding));
        choicesDiv.appendChild(btn);
    });
}

    buffer = parts.pop();

    if (buffer.trim()) {
        explanation += buffer;
        res.write(`data: ${JSON.stringify({ paragraph: buffer })}\n\n`);
    }

    // Send the final answer explicitly
    res.write(`data: ${JSON.stringify({
        solution,
        explanation,                                    answer: solution, // <-- This is what the frontend needs
        done: true
    })}\n\n`);

    res.end();
});
